<template>
    <div>
      <p>글씨체</p>
      <hr>
      <p>글씨 색상</p>
      <div class="btn-group">
    <button
      v-for="color in colors"
      :key="color.name"
      :class="['btn', `btn-${color.name}`, { 'btn-selected': selectedColor === color.name }]"
      @click="selectColor(color.name)"
    >
      {{ color.label }}
    </button>
  </div>
    </div>
  </template>
  
  <script setup>
  import { ref } from 'vue';
  
  const emit = defineEmits(['colorSelected']);
  const selectedColor = ref('');
  
  const colors = [
    { name: 'black', label: '검정' },
    { name: 'blue', label: '파랑' },
    { name: 'purple', label: '보라' },
    { name: 'green', label: '초록' },
    { name: 'navy', label: '네이비' }
  ];
  
  const selectColor = (color) => {
    selectedColor.value = color;
    console.log(color); // 색상 잘 찍힘
    emit('colorSelected', color);
  };
  </script>
  
  <style>
  .btn {
    /* 공통 버튼 스타일 */
    margin: 5px;
    padding: 10px 20px;
    border: none;
    cursor: pointer;
    font-size: 16px;
  }

  .btn-selected {
    border: 2px solid #ff4500; /* 하이라이트 표시 */
    font-weight: bold; /* 선택된 버튼의 글씨 굵게 */
  }
  
.btn-black { background-color: black;}
.btn-blue { background-color: blue;}
.btn-purple { background-color: purple;}
.btn-green { background-color: green;}
.btn-navy { background-color: navy;}
</style>
